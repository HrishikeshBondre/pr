#include <stdio.h>
#include <string.h>

int main() {
    char text[100];
    int key;

    printf("Msg: ");
    scanf("%s",text);
    printf("Key: ");
    scanf("%d",&key);

    int len = strlen(text);
    char rail[key][len];
    for(int i =0; i<key; i++){
        for(int j = 0; j<len; j++){
            rail[i][j] = '\n';
        }
    }

    int row = 0, down =1;
    for(int i = 0; i<len; i++){
        rail[row][i] = text[i];
        if(row==0){
            down = 1;
        }
        else if(row == key -1){
            down = 0;
        }

        row += (down ? 1: -1);
    }

    for(int i =0; i<key; i++){
        for(int j =0; j<len; j++){
            if(rail[i][j] != '\n'){
                printf("%c", rail[i][j]);
            }
        }
    }

    return 0;
}

#include <stdio.h>
#include <string.h>
int main() {
    char text[500], key[50];
    printf("Enter message: ");
    scanf("%s", text);
    printf("Enter key: ");
    scanf("%s", key);
    int len = strlen(text);
    int keylen = strlen(key);
    int rows = len / keylen;
    if (len % keylen != 0)
        rows++;
    char matrix[rows][keylen];
    int k = 0;
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < keylen; j++) {
            if (k < len)
                matrix[i][j] = text[k++];
            else
                matrix[i][j] = 'X';
        }
    }
    int order[50];
    for (int i = 0; i < keylen; i++) {
        order[i] = i;
    }
    for (int i = 0; i < keylen - 1; i++) {
        for (int j = i + 1; j < keylen; j++) {
            if (key[order[i]] > key[order[j]]) {
                int temp = order[i];
                order[i] = order[j];
                order[j] = temp;
            }
        }
    }
    printf("Encrypted Message: ");
    for (int x = 0; x < keylen; x++) {
        int col = order[x];
        for (int i = 0; i < rows; i++) {
            printf("%c", matrix[i][col]);
        }
    }
    return 0;
}

#include <stdio.h>
#include <string.h>

long long mod_pow(long long base, long long exp, long long n) {
    long long result = 1;
    for (int i = 0; i < exp; i++)
        result = (result * base) % n;
    return result;
}

int main() {
    long long p = 3, q = 11;
    long long n = p * q;
    long long e = 7;
    long long d = 3;
    char text[100];

    printf("Enter text (A-Z only): ");
    fgets(text, sizeof(text), stdin);
    text[strcspn(text, "\n")] = 0;

    printf("\nEncrypted: ");
    for (int i = 0; i < strlen(text); i++) {
        if (text[i] >= 'A' && text[i] <= 'Z') {
            printf("%lld ", mod_pow(text[i] - 'A', e, n));
        } else {
            printf("%c ", text[i]);
        }
    }

    printf("\nDecrypted: ");
    for (int i = 0; i < strlen(text); i++) {
        if (text[i] >= 'A' && text[i] <= 'Z') {
            printf("%c", (char)(mod_pow(mod_pow(text[i] - 'A', e, n), d, n) + 'A'));
        } else {
            printf("%c", text[i]);
        }
    }

    printf("\n");
    return 0;
}

#include <stdio.h>

int main() {
    int primeNum, root;
    int privAlice, privBob;
    int pubAlice = 1, pubBob = 1;
    int i, sharedA = 1, sharedB = 1;
    
    printf("Enter prime: ");
    scanf("%d", &primeNum);
    printf("Enter root: ");
    scanf("%d", &root);
    
    printf("Enter alice priv: ");
    scanf("%d", &privAlice);
    printf("Enter bob priv: ");
    scanf("%d", &privBob);
    
    for (i = 0; i < privAlice; i++) {
        pubAlice = (pubAlice * root) % primeNum;
    }
    for (i = 0; i < privBob; i++) {
        pubBob = (pubBob * root) % primeNum;
    }
    
    printf("Alice pub: %d", pubAlice);
    printf("Bob pub: %d", pubBob);
    
    for (i = 0; i < privAlice; i++) {
        sharedA = (sharedA * pubBob) % primeNum;
    }
    for (i = 0; i < privBob; i++) {
        sharedB = (sharedB * pubAlice) % primeNum;
    }
    
    printf("Alice : %d", sharedA);
    printf("Bob : %d", sharedB);
    
    return 0;
}

#include <stdio.h>

int main() {
    FILE *file;
    int ch;
    
    file = fopen("log.text", "a");
    if (file == NULL) {
        perror("Error opening file");
        return 1;
    }
    
    printf("Start typing: ");
    
    while ((ch = getchar()) != EOF) {
        if (ch == 'O') {
            break;
        }
        fputc(ch, file);
    }
    
    if (fclose(file) != 0) {
        perror("File not closing");
        return 1;
    }
    
    printf("Typing stopped");
    
    return 0;
}
